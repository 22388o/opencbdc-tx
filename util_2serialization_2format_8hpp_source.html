108 auto operator>>(serializer& deser, std::optional<T>& val) -> serializer& {
109 bool has_value{};
110 if(!(deser >> has_value)) {
111 return deser;
112 }
113
114 if(has_value) {
115 auto opt_val = T();
116 if(!(deser >> opt_val)) {

121 val = std::nullopt;

124 return deser;

186 template<typename T>
187 auto operator>>(serializer& packet, std::vector<T>& vec) -> serializer& {
188 static_assert(sizeof(T) <= config::maximum_reservation,
189 "Vector element size too large");
190
191 uint64_t len{};
192 if(!(packet >> len)) {
193 return packet;
194 }
195
196 uint64_t allocated = 0;
197 while(allocated < len) {
198 allocated = std::min(
199 len,
200 allocated + config::maximum_reservation / sizeof(T));
201 vec.reserve(allocated);
202 while(vec.size() < allocated) {
203 if constexpr(std::is_default_constructible_v<T>) {
204 T val{};
205 if(!(packet >> val)) {
206 return packet;
207 }
208 vec.push_back(std::move(val));
209 } else {
210 auto val = T(packet);
211 if(!packet) {
212 return packet;
213 }
214 vec.push_back(std::move(val));
215 }
216 }
217 }
218
219 vec.shrink_to_fit();
220 return packet;
221 }

241 template<typename K, typename V, typename... Ts>
242 auto operator>>(serializer& deser, std::unordered_map<K, V, Ts...>& map)
243 -> serializer& {
244 static_assert(sizeof(K) + sizeof(V) <= config::maximum_reservation,
245 "Unordered Map element size too large");
246 auto len = uint64_t();
247 if(!(deser >> len)) {
248 return deser;
249 }
250
251 uint64_t allocated = 0;
252 while(allocated < len) {
253 allocated = std::min(len,
254 allocated
255 + config::maximum_reservation
256 / (sizeof(K) + sizeof(V)));
257 map.reserve(allocated);
258 while(map.size() < allocated) {
259 auto key = K();
260 if(!(deser >> key)) {
261 return deser;
262 }
263
264 auto val = V();
265 if(!(deser >> val)) {
266 return deser;
267 }
268
269 map.emplace(std::move(key), std::move(val));
270 }
271 }
272 return deser;
273 }

351 template<typename... Ts>
352 auto operator<<(serializer& ser, const std::variant<Ts...>& var)
353 -> serializer& {
354 using S = uint8_t;
355 static_assert(
356 std::variant_size_v<std::remove_reference_t<decltype(var)>> < std::
357 numeric_limits<S>::max());
358 auto idx = static_cast<S>(var.index());
359 ser << idx;
360 std::visit(
361 [&](auto&& arg) {
362 ser << arg;
363 },
364 var);
365 return ser;
366 }

370 template<typename... Ts>
371 auto operator>>(serializer& deser, std::variant<Ts...>& var)
372 -> std::enable_if_t<(std::is_default_constructible_v<Ts> && ...),
373 serializer&> {
374 using S = uint8_t;
375 static_assert(
376 std::variant_size_v<std::remove_reference_t<decltype(var)>> < std::

396 template<typename... Ts>
397 [[nodiscard]] auto get_variant(serializer& deser) -> std::variant<Ts...> {
398 using S = uint8_t;
399 using T = typename std::variant<Ts...>;
400 static_assert(std::variant_size_v<T> < std::numeric_limits<S>::max());
401 S idx{};
402 deser >> idx;
403 auto i = static_cast<size_t>(idx);
404 assert(i < std::variant_size_v<T>);
405 static constexpr auto t = std::array{+[](serializer& d) {
406 return T{std::in_place_type<Ts>, d};
407 }...};
408 // TODO: deserialization error handling for variant indexes.
409 return t.at(i)(deser);
410 }
